144     /**
145      * Mapping from HTTP method annotation to {@link HttpMethod}, like following.
146      * <ul>
147      *   <li>{@link Options} -> {@link HttpMethod#OPTIONS}
148      *   <li>{@link Get} -> {@link HttpMethod#GET}
149      *   <li>{@link Head} -> {@link HttpMethod#HEAD}
150      *   <li>{@link Post} -> {@link HttpMethod#POST}
151      *   <li>{@link Put} -> {@link HttpMethod#PUT}
152      *   <li>{@link Patch} -> {@link HttpMethod#PATCH}
153      *   <li>{@link Delete} -> {@link HttpMethod#DELETE}
154      *   <li>{@link Trace} -> {@link HttpMethod#TRACE}
155      * </ul>
156      */
157     private static final Map<Class<?>, HttpMethod> HTTP_METHOD_MAP =
158             ImmutableMap.<Class<?>, HttpMethod>builder()
159                     .put(Options.class, HttpMethod.OPTIONS)
160                     .put(Get.class, HttpMethod.GET)
161                     .put(Head.class, HttpMethod.HEAD)
162                     .put(Post.class, HttpMethod.POST)
163                     .put(Put.class, HttpMethod.PUT)
164                     .put(Patch.class, HttpMethod.PATCH)
165                     .put(Delete.class, HttpMethod.DELETE)
166                     .put(Trace.class, HttpMethod.TRACE)
167                     .build();

169     /**
170      * Returns the list of {@link AnnotatedHttpService} defined by {@link Path} and HTTP method annotations
171      * from the specified {@code object}.
172      */
173     public static List<AnnotatedHttpServiceElement> find(String pathPrefix, Object object,
174                                                          Iterable<?> exceptionHandlersAndConverters) {
175         Builder<ExceptionHandlerFunction> exceptionHandlers = null;
176         Builder<RequestConverterFunction> requestConverters = null;
177         Builder<ResponseConverterFunction> responseConverters = null;
178 
179         for (final Object o : exceptionHandlersAndConverters) {
180             boolean added = false;
181             if (o instanceof ExceptionHandlerFunction) {
182                 if (exceptionHandlers == null) {
183                     exceptionHandlers = ImmutableList.builder();
184                 }
185                 exceptionHandlers.add((ExceptionHandlerFunction) o);
186                 added = true;
187             }
188             if (o instanceof RequestConverterFunction) {
189                 if (requestConverters == null) {
190                     requestConverters = ImmutableList.builder();
191                 }
192                 requestConverters.add((RequestConverterFunction) o);
193                 added = true;
194             }
195             if (o instanceof ResponseConverterFunction) {
196                 if (responseConverters == null) {
197                     responseConverters = ImmutableList.builder();
198                 }
199                 responseConverters.add((ResponseConverterFunction) o);
200                 added = true;
201             }
202             if (!added) {
203                 throw new IllegalArgumentException(o.getClass().getName() +
204                                                    " is neither an exception handler nor a converter.");
205             }
206         }
207 
208         final List<ExceptionHandlerFunction> exceptionHandlerFunctions =
209                 exceptionHandlers != null ? exceptionHandlers.build() : ImmutableList.of();
210         final List<RequestConverterFunction> requestConverterFunctions =
211                 requestConverters != null ? requestConverters.build() : ImmutableList.of();
212         final List<ResponseConverterFunction> responseConverterFunctions =
213                 responseConverters != null ? responseConverters.build() : ImmutableList.of();
214 
215         final List<Method> methods = requestMappingMethods(object);
216         return methods.stream()
217                       .map((Method method) -> create(pathPrefix, object, method, exceptionHandlerFunctions,
218                                                      requestConverterFunctions, responseConverterFunctions))
219                       .collect(toImmutableList());
220     }

222     private static HttpStatus defaultResponseStatus(Optional<HttpStatus> defaultResponseStatus,
223                                                     Method method) {
224         return defaultResponseStatus.orElseGet(() -> {
225             // Set a default HTTP status code for a response depending on the return type of the method.
226             final Class<?> returnType = method.getReturnType();
227             return returnType == Void.class ||
228                    returnType == void.class ? HttpStatus.NO_CONTENT : HttpStatus.OK;
229         });
230     }

232     private static <T extends Annotation> void setAdditionalHeader(HttpHeaders headers,
233                                                                    AnnotatedElement element,
234                                                                    String clsAlias,
235                                                                    String elementAlias,
236                                                                    String level,
237                                                                    Class<T> annotation,
238                                                                    Function<T, String> nameGetter,
239                                                                    Function<T, String[]> valueGetter) {
240         requireNonNull(headers, "headers");
241         requireNonNull(element, "element");
242         requireNonNull(level, "level");
243 
244         final Set<String> addedHeaderSets = new HashSet<>();
245         findAll(element, annotation).forEach(header -> {
246             final String name = nameGetter.apply(header);
247             final String[] value = valueGetter.apply(header);
248 
249             if (addedHeaderSets.contains(name)) {
250                 logger.warn("The additional {} named '{}' at '{}' is set at the same {} level already;" +
251                             "ignoring.",
252                             clsAlias, name, elementAlias, level);
253                 return;
254             }
255             headers.set(HttpHeaderNames.of(name), value);
256             addedHeaderSets.add(name);
257         });
258     }

260     /**
261      * Returns an {@link AnnotatedHttpService} instance defined to {@code method} of {@code object} using
262      * {@link Path} annotation.
263      */
264     private static AnnotatedHttpServiceElement create(String pathPrefix, Object object, Method method,
265                                                       List<ExceptionHandlerFunction> baseExceptionHandlers,
266                                                       List<RequestConverterFunction> baseRequestConverters,
267                                                       List<ResponseConverterFunction> baseResponseConverters) {
268 
269         final Set<Annotation> methodAnnotations = httpMethodAnnotations(method);
270         if (methodAnnotations.isEmpty()) {
271             throw new IllegalArgumentException("HTTP Method specification is missing: " + method.getName());
272         }
273 
274         final Set<HttpMethod> methods = toHttpMethods(methodAnnotations);
275         if (methods.isEmpty()) {
276             throw new IllegalArgumentException(method.getDeclaringClass().getName() + '#' + method.getName() +
277                                                " must have an HTTP method annotation.");
278         }
279 
280         final Class<?> clazz = object.getClass();
281         final PathMapping pathMapping = pathStringMapping(pathPrefix, method, methodAnnotations)
282                 .withHttpHeaderInfo(methods, consumableMediaTypes(method, clazz),
283                                     producibleMediaTypes(method, clazz));
284 
285         final List<ExceptionHandlerFunction> eh =
286                 getAnnotatedInstances(method, clazz, ExceptionHandler.class, ExceptionHandlerFunction.class)
287                         .addAll(baseExceptionHandlers).add(defaultExceptionHandler).build();
288         final List<RequestConverterFunction> req =
289                 getAnnotatedInstances(method, clazz, RequestConverter.class, RequestConverterFunction.class)
290                         .addAll(baseRequestConverters).build();
291         final List<ResponseConverterFunction> res =
292                 getAnnotatedInstances(method, clazz, ResponseConverter.class, ResponseConverterFunction.class)
293                         .addAll(baseResponseConverters).build();
294 
295         List<AnnotatedValueResolver> resolvers;
296         try {
297             resolvers = AnnotatedValueResolver.ofServiceMethod(method, pathMapping.paramNames(),
298                                                                toRequestObjectResolvers(req));
299         } catch (NoParameterException ignored) {
300             // Allow no parameter like below:
301             //
302             // @Get("/")
303             // public String method1() { ... }
304             //
305             resolvers = ImmutableList.of();
306         }
307 
308         final Set<String> expectedParamNames = pathMapping.paramNames();
309         final Set<String> requiredParamNames =
310                 resolvers.stream()
311                          .filter(AnnotatedValueResolver::isPathVariable)
312                          .map(AnnotatedValueResolver::httpElementName)
313                          .collect(Collectors.toSet());
314 
315         if (!expectedParamNames.containsAll(requiredParamNames)) {
316             final Set<String> missing = Sets.difference(requiredParamNames, expectedParamNames);
317             throw new IllegalArgumentException("cannot find path variables: " + missing);
318         }
319 
320         // Warn unused path variables only if there's no '@RequestObject' annotation.
321         if (resolvers.stream().noneMatch(r -> r.annotationType() == RequestObject.class) &&
322             !requiredParamNames.containsAll(expectedParamNames)) {
323             final Set<String> missing = Sets.difference(expectedParamNames, requiredParamNames);
324             logger.warn("Some path variables of the method '" + method.getName() +
325                         "' of the class '" + clazz.getName() +
326                         "' do not have their corresponding parameters annotated with @Param. " +
327                         "They would not be automatically injected: " + missing);
328         }
329 
330         final Optional<HttpStatus> defaultResponseStatus = findFirst(method, StatusCode.class)
331                 .map(code -> {
332                     final int statusCode = code.value();
333                     checkArgument(statusCode >= 0,
334                                   "invalid HTTP status code: %s (expected: >= 0)", statusCode);
335                     return HttpStatus.valueOf(statusCode);
336                 });
337         final HttpHeaders defaultHeaders = HttpHeaders.of(defaultResponseStatus(defaultResponseStatus, method));
338 
339         final HttpHeaders defaultTrailingHeaders = HttpHeaders.of();
340         final String classAlias = clazz.getName();
341         final String methodAlias = String.format("%s.%s()", classAlias, method.getName());
342         setAdditionalHeader(defaultHeaders, clazz, "header", classAlias, "class", AdditionalHeader.class,
343                             AdditionalHeader::name, AdditionalHeader::value);
344         setAdditionalHeader(defaultHeaders, method, "header", methodAlias, "method", AdditionalHeader.class,
345                             AdditionalHeader::name, AdditionalHeader::value);
346         setAdditionalHeader(defaultTrailingHeaders, clazz, "trailer", classAlias, "class",
347                             AdditionalTrailer.class, AdditionalTrailer::name, AdditionalTrailer::value);
348         setAdditionalHeader(defaultTrailingHeaders, method, "trailer", methodAlias, "method",
349                             AdditionalTrailer.class, AdditionalTrailer::name, AdditionalTrailer::value);
350 
351         if (ArmeriaHttpUtil.isContentAlwaysEmpty(defaultHeaders.status()) &&
352             !defaultTrailingHeaders.isEmpty()) {
353             logger.warn("A response with HTTP status code '{}' cannot have a content. " +
354                         "Trailing headers defined at '{}' might be ignored.",
355                         defaultHeaders.status().code(), methodAlias);
356         }
357 
358         // A CORS preflight request can be received because we handle it specially. The following
359         // decorator will prevent the service from an unexpected request which has OPTIONS method.
360         final Function<Service<HttpRequest, HttpResponse>,
361                 ? extends Service<HttpRequest, HttpResponse>> initialDecorator;
362         if (methods.contains(HttpMethod.OPTIONS)) {
363             initialDecorator = Function.identity();
364         } else {
365             initialDecorator = delegate -> new SimpleDecoratingService<HttpRequest, HttpResponse>(delegate) {
366                 @Override
367                 public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
368                     if (req.method() == HttpMethod.OPTIONS) {
369                         // This must be a CORS preflight request.
370                         throw HttpStatusException.of(HttpStatus.FORBIDDEN);
371                     }
372                     return delegate().serve(ctx, req);
373                 }
374             };
375         }
376         return new AnnotatedHttpServiceElement(pathMapping,
377                                                new AnnotatedHttpService(object, method, resolvers, eh,
378                                                                         res, pathMapping, defaultHeaders,
379                                                                         defaultTrailingHeaders),
380                                                decorator(method, clazz, initialDecorator));
381     }

383     /**
384      * Returns the list of {@link Path} annotated methods.
385      */
386     private static List<Method> requestMappingMethods(Object object) {
387         return getAllMethods(object.getClass(), withModifier(Modifier.PUBLIC))
388                 .stream()
389                 // Lookup super classes just in case if the object is a proxy.
390                 .filter(m -> getAnnotations(m, FindOption.LOOKUP_SUPER_CLASSES)
391                         .stream()
392                         .map(Annotation::annotationType)
393                         .anyMatch(a -> a == Path.class ||
394                                        HTTP_METHOD_MAP.containsKey(a)))
395                 .sorted(Comparator.comparingInt(AnnotatedHttpServiceFactory::order))
396                 .collect(toImmutableList());
397     }

399     /**
400      * Returns the value of the order of the {@link Method}. The order could be retrieved from {@link Order}
401      * annotation. 0 would be returned if there is no specified {@link Order} annotation.
402      */
403     private static int order(Method method) {
404         final Order order = findFirst(method, Order.class).orElse(null);
405         return order != null ? order.value() : 0;

429      * Returns {@link Set} of {@link HttpMethod}s mapped to HTTP method annotations.
430      *
431      * @see Options
432      * @see Get
433      * @see Head
434      * @see Post
435      * @see Put

447     /**
448      * Returns the list of {@link MediaType}s specified by {@link Consumes} annotation.
449      */
450     private static List<MediaType> consumableMediaTypes(Method method, Class<?> clazz) {
451         List<Consumes> consumes = findAll(method, Consumes.class);
452         List<ConsumeType> consumeTypes = findAll(method, ConsumeType.class);
453 
454         if (consumes.isEmpty() && consumeTypes.isEmpty()) {
455             consumes = findAll(clazz, Consumes.class);
456             consumeTypes = findAll(clazz, ConsumeType.class);
457         }
458 
459         final List<MediaType> types =
460                 Stream.concat(consumes.stream().map(Consumes::value),
461                               consumeTypes.stream().map(ConsumeType::value))
462                       .map(MediaType::parse)
463                       .collect(toImmutableList());
464         return ensureUniqueTypes(types, Consumes.class);
465     }
466 
467     /**
468      * Returns the list of {@link MediaType}s specified by {@link Produces} annotation.
469      */
470     private static List<MediaType> producibleMediaTypes(Method method, Class<?> clazz) {
471         List<Produces> produces = findAll(method, Produces.class);
472         List<ProduceType> produceTypes = findAll(method, ProduceType.class);
473 
474         if (produces.isEmpty() && produceTypes.isEmpty()) {
475             produces = findAll(clazz, Produces.class);
476             produceTypes = findAll(clazz, ProduceType.class);
477         }
478 
479         final List<MediaType> types =
480                 Stream.concat(produces.stream().map(Produces::value),
481                               produceTypes.stream().map(ProduceType::value))
482                       .map(MediaType::parse)
483                       .peek(type -> {
484                           if (type.hasWildcard()) {
485                               throw new IllegalArgumentException(
486                                       "Producible media types must not have a wildcard: " + type);
487                           }
488                       })
489                       .collect(toImmutableList());
490         return ensureUniqueTypes(types, Produces.class);
491     }
492 
493     private static List<MediaType> ensureUniqueTypes(List<MediaType> types, Class<?> annotationClass) {
494         final Set<MediaType> set = new HashSet<>();
495         for (final MediaType type : types) {
496             if (!set.add(type)) {
497                 throw new IllegalArgumentException(
498                         "Duplicated media type for @" + annotationClass.getSimpleName() + ": " + type);
499             }
500         }
501         return types;
502     }

504     /**
505      * Returns the {@link PathMapping} instance mapped to {@code method}.
506      */
507     private static PathMapping pathStringMapping(String pathPrefix, Method method,
508                                                  Set<Annotation> methodAnnotations) {
509         pathPrefix = ensureAbsolutePath(pathPrefix, "pathPrefix");
510         if (!pathPrefix.endsWith("/")) {
511             pathPrefix += '/';
512         }
513 
514         final String pattern = findPattern(method, methodAnnotations);
515         final PathMapping mapping = PathMapping.of(pattern);
516         if ("/".equals(pathPrefix)) {
517             // pathPrefix is not specified or "/".
518             return mapping;
519         }
520 
521         if (pattern.startsWith(EXACT)) {
522             return PathMapping.ofExact(concatPaths(
523                     pathPrefix, pattern.substring(EXACT.length())));
524         }
525 
526         if (pattern.startsWith(PREFIX)) {
527             return PathMapping.ofPrefix(concatPaths(
528                     pathPrefix, pattern.substring(PREFIX.length())));
529         }
530 
531         if (pattern.startsWith(GLOB)) {
532             final String glob = pattern.substring(GLOB.length());
533             if (glob.startsWith("/")) {
534                 return PathMapping.ofGlob(concatPaths(pathPrefix, glob));
535             } else {
536                 // NB: We cannot use PathMapping.ofGlob(pathPrefix + "/**/" + glob) here
537                 //     because that will extract '/**/' as a path parameter, which a user never specified.
538                 return new PrefixAddingPathMapping(pathPrefix, mapping);
539             }
540         }
541 
542         if (pattern.startsWith(REGEX)) {
543             return new PrefixAddingPathMapping(pathPrefix, mapping);
544         }
545 
546         if (pattern.startsWith("/")) {
547             // Default pattern
548             return PathMapping.of(concatPaths(pathPrefix, pattern));
549         }
550 
551         // Should never reach here because we validated the path pattern.
552         throw new Error();
553     }

555     /**
556      * Returns a specified path pattern. The path pattern might be specified by {@link Path} or
557      * HTTP method annotations such as {@link Get} and {@link Post}.
558      */
559     private static String findPattern(Method method, Set<Annotation> methodAnnotations) {
560         String pattern = findFirst(method, Path.class).map(Path::value)
561                                                       .orElse(null);
562         for (Annotation a : methodAnnotations) {
563             final String p = (String) invokeValueMethod(a);
564             if (DefaultValues.isUnspecified(p)) {
565                 continue;
566             }
567             checkArgument(pattern == null,
568                           "Only one path can be specified. (" + pattern + ", " + p + ')');
569             pattern = p;
570         }
571         if (pattern == null || pattern.isEmpty()) {
572             throw new IllegalArgumentException(
573                     "A path pattern should be specified by @Path or HTTP method annotations.");
574         }
575         return pattern;
576     }

582     private static Function<Service<HttpRequest, HttpResponse>,

584             Method method, Class<?> clazz,
585             Function<Service<HttpRequest, HttpResponse>,
586                     ? extends Service<HttpRequest, HttpResponse>> initialDecorator) {
587 
588         final List<DecoratorAndOrder> decorators = collectDecorators(clazz, method);
589 
590         Function<Service<HttpRequest, HttpResponse>,
591                 ? extends Service<HttpRequest, HttpResponse>> decorator = initialDecorator;
592         for (int i = decorators.size() - 1; i >= 0; i--) {
593             final DecoratorAndOrder d = decorators.get(i);
594             decorator = decorator.andThen(d.decorator());
595         }
596         return decorator;
597     }

599     /**
600      * Returns a decorator list which is specified by {@link Decorator} annotations and user-defined
601      * decorator annotations.
602      */
603     @VisibleForTesting
604     static List<DecoratorAndOrder> collectDecorators(Class<?> clazz, Method method) {
605         final List<DecoratorAndOrder> decorators = new ArrayList<>();
606 
607         // Class-level decorators are applied before method-level decorators.
608         collectDecorators(decorators, getAllAnnotations(clazz));
609         collectDecorators(decorators, getAllAnnotations(method));
610 
611         // Sort decorators by "order" attribute values.
612         decorators.sort(Comparator.comparing(DecoratorAndOrder::order));
613 
614         return decorators;
615     }

617     /**
618      * Adds decorators to the specified {@code list}. Decorators which are annotated with {@link Decorator}
619      * and user-defined decorators will be collected.
620      */
621     private static void collectDecorators(List<DecoratorAndOrder> list, List<Annotation> annotations) {
622         if (annotations.isEmpty()) {
623             return;
624         }
625 
626         // Respect the order of decorators which is specified by a user. The first one is first applied
627         // for most of the cases. But if @Decorator and user-defined decorators are specified in a mixed order,
628         // the specified order and the applied order can be different. To overcome this problem, we introduce
629         // "order" attribute to @Decorator annotation to sort decorators. If a user-defined decorator
630         // annotation has "order" attribute, it will be also used for sorting.
631         for (final Annotation annotation : annotations) {
632             if (annotation instanceof Decorator) {
633                 final Decorator d = (Decorator) annotation;
634                 list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
635                 continue;
636             }
637 
638             if (annotation instanceof Decorators) {
639                 final Decorator[] decorators = ((Decorators) annotation).value();
640                 for (final Decorator d : decorators) {
641                     list.add(new DecoratorAndOrder(d, newDecorator(d), d.order()));
642                 }
643                 continue;
644             }
645 
646             DecoratorAndOrder udd = userDefinedDecorator(annotation);
647             if (udd != null) {
648                 list.add(udd);
649                 continue;
650             }
651 
652             // If user-defined decorators are repeatable and they are specified more than once.
653             try {
654                 final Method method = Iterables.getFirst(getMethods(annotation.annotationType(),
655                                                                     withName("value")), null);
656                 assert method != null : "No 'value' method is found from " + annotation;
657                 final Annotation[] decorators = (Annotation[]) method.invoke(annotation);
658                 for (final Annotation decorator : decorators) {
659                     udd = userDefinedDecorator(decorator);
660                     if (udd == null) {
661                         break;
662                     }
663                     list.add(udd);
664                 }
665             } catch (Throwable ignore) {
666                 // The annotation may be a container of a decorator or may be not, so we just ignore
667                 // any exception from this clause.
668             }
669         }
670     }

672     /**
673      * Returns a decorator with its order if the specified {@code annotation} is one of the user-defined
674      * decorator annotation.
675      */
676     @Nullable
677     private static DecoratorAndOrder userDefinedDecorator(Annotation annotation) {
678         // User-defined decorator MUST be annotated with @DecoratorFactory annotation.
679         final DecoratorFactory d =
680                 findFirstDeclared(annotation.annotationType(), DecoratorFactory.class).orElse(null);
681         if (d == null) {
682             return null;
683         }
684 
685         // In case of user-defined decorator, we need to create a new decorator from its factory.
686         @SuppressWarnings("unchecked")
687         final DecoratorFactoryFunction<Annotation> factory = getInstance(d, DecoratorFactoryFunction.class);
688 
689         // If the annotation has "order" attribute, we can use it when sorting decorators.
690         int order = 0;
691         try {
692             final Method method = Iterables.getFirst(getMethods(annotation.annotationType(),
693                                                                 withName("order")), null);
694             if (method != null) {
695                 final Object value = method.invoke(annotation);
696                 if (value instanceof Integer) {
697                     order = (Integer) value;
698                 }
699             }
700         } catch (Throwable ignore) {
701             // A user-defined decorator may not have an 'order' attribute.
702             // If it does not exist, '0' is used by default.
703         }
704         return new DecoratorAndOrder(annotation, factory.newDecorator(annotation), order);
705     }

717     /**
718      * Returns a {@link Builder} which has the instances specified by the annotations of the
719      * {@code annotationType}. The annotations of the specified {@code method} and {@code clazz} will be
720      * collected respectively.
721      */
722     private static <T extends Annotation, R> Builder<R> getAnnotatedInstances(
723             AnnotatedElement method, AnnotatedElement clazz, Class<T> annotationType, Class<R> resultType) {
724         final Builder<R> builder = new Builder<>();
725         Stream.concat(findAll(method, annotationType).stream(),
726                       findAll(clazz, annotationType).stream())
727               .forEach(annotation -> builder.add(getInstance(annotation, resultType)));
728         return builder;

731     /**
732      * Returns a cached instance of the specified {@link Class} which is specified in the given
733      * {@link Annotation}.
734      */
735     static <T> T getInstance(Annotation annotation, Class<T> expectedType) {
736         try {
737             @SuppressWarnings("unchecked")
738             final Class<? extends T> clazz = (Class<? extends T>) invokeValueMethod(annotation);
739             return expectedType.cast(instanceCache.computeIfAbsent(clazz, type -> {
740                 try {
741                     return getInstance0(clazz);
742                 } catch (Exception e) {
743                     throw new IllegalStateException(
744                             "A class specified in @" + annotation.getClass().getSimpleName() +
745                             " annotation must have an accessible default constructor: " + clazz.getName(), e);
746                 }
747             }));
748         } catch (ClassCastException e) {
749             throw new IllegalArgumentException(
750                     "A class specified in @" + annotation.getClass().getSimpleName() +
751                     " annotation cannot be cast to " + expectedType, e);
752         }
753     }

755     /**
756      * Returns a cached instance of the specified {@link Class}.
757      */
758     static <T> T getInstance(Class<T> clazz) {
759         @SuppressWarnings("unchecked")
760         final T casted = (T) instanceCache.computeIfAbsent(clazz, type -> {
761             try {
762                 return getInstance0(clazz);
763             } catch (Exception e) {
764                 throw new IllegalStateException("A class must have an accessible default constructor: " +
765                                                 clazz.getName(), e);
766             }
767         });
768         return casted;
769     }
770 
771     private static <T> T getInstance0(Class<? extends T> clazz) throws Exception {
772         @SuppressWarnings("unchecked")
773         final Constructor<? extends T> constructor =
774                 Iterables.getFirst(getConstructors(clazz, withParametersCount(0)), null);
775         assert constructor != null : "No default constructor is found from " + clazz.getName();
776         constructor.setAccessible(true);
777         return constructor.newInstance();
778     }

780     /**
781      * Returns an object which is returned by {@code value()} method of the specified annotation {@code a}.
782      */
783     private static Object invokeValueMethod(Annotation a) {
784         try {
785             final Method method = Iterables.getFirst(getMethods(a.getClass(), withName("value")), null);
786             assert method != null : "No 'value' method is found from " + a;
787             return method.invoke(a);
788         } catch (Exception e) {
789             throw new IllegalStateException("An annotation @" + a.getClass().getSimpleName() +
790                                             " must have a 'value' method", e);
791         }
792     }

795      * Returns the description of the specified {@link AnnotatedElement}.
796      */
797     @Nullable
798     static String findDescription(AnnotatedElement annotatedElement) {
799         requireNonNull(annotatedElement, "annotatedElement");
800         final Optional<Description> description = findFirst(annotatedElement, Description.class);
801         if (description.isPresent()) {
802             final String value = description.get().value();
803             if (DefaultValues.isSpecified(value)) {
804                 checkArgument(!value.isEmpty(), "value is empty");

806             }
807         }
808         return null;

813     /**
814      * A {@link PathMapping} implementation that combines path prefix and another {@link PathMapping}.
815      */
816     @VisibleForTesting
817     static final class PrefixAddingPathMapping extends AbstractPathMapping {
818 
819         private final String pathPrefix;
820         private final PathMapping mapping;
821         private final String loggerName;
822         private final String meterTag;
823 
824         PrefixAddingPathMapping(String pathPrefix, PathMapping mapping) {
825             requireNonNull(mapping, "mapping");
826             // mapping should be GlobPathMapping or RegexPathMapping
827             assert mapping.regex().isPresent() : "unexpected mapping type: " + mapping.getClass().getName();
828             this.pathPrefix = requireNonNull(pathPrefix, "pathPrefix");
829             this.mapping = mapping;
830             loggerName = newLoggerName(pathPrefix) + '.' + mapping.loggerName();
831             meterTag = PREFIX + pathPrefix + ',' + mapping.meterTag();
832         }
833 
834         @Override
835         protected PathMappingResult doApply(PathMappingContext mappingCtx) {
836             final String path = mappingCtx.path();
837             if (!path.startsWith(pathPrefix)) {
838                 return PathMappingResult.empty();
839             }
840 
841             final PathMappingResult result =
842                     mapping.apply(mappingCtx.overridePath(path.substring(pathPrefix.length() - 1)));
843             if (result.isPresent()) {
844                 return PathMappingResult.of(path, mappingCtx.query(), result.pathParams());
845             } else {
846                 return PathMappingResult.empty();
847             }
848         }
849 
850         @Override
851         public Set<String> paramNames() {
852             return mapping.paramNames();
853         }
854 
855         @Override
856         public String loggerName() {
857             return loggerName;
858         }
859 
860         @Override
861         public String meterTag() {
862             return meterTag;
863         }
864 
865         @Override
866         public Optional<String> prefix() {
867             return Optional.of(pathPrefix);
868         }
869 
870         @Override
871         public Optional<String> regex() {
872             return mapping.regex();
873         }
874 
875         @Override
876         public boolean equals(Object o) {
877             if (this == o) {
878                 return true;
879             }
880             if (!(o instanceof PrefixAddingPathMapping)) {
881                 return false;
882             }
883 
884             final PrefixAddingPathMapping that = (PrefixAddingPathMapping) o;
885             return pathPrefix.equals(that.pathPrefix) && mapping.equals(that.mapping);
886         }
887 
888         @Override
889         public int hashCode() {
890             return 31 * pathPrefix.hashCode() + mapping.hashCode();
891         }
892 
893         @Override
894         public String toString() {
895             return '[' + PREFIX + pathPrefix + ", " + mapping + ']';
896         }
897     }
