17  package com.linecorp.armeria.common.logging;
18  
19  import static com.google.common.base.Preconditions.checkArgument;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.util.function.Function;
23  
24  import com.google.common.base.MoreObjects;
25  import com.google.common.base.MoreObjects.ToStringHelper;
26  
27  import com.linecorp.armeria.common.HttpHeaders;
28  
29  /**
30   * Builds a new logging decorator.
31   */
32  public abstract class LoggingDecoratorBuilder<T extends LoggingDecoratorBuilder<T>> {
33      private static final Function<HttpHeaders, HttpHeaders> DEFAULT_HEADERS_SANITIZER = Function.identity();
34      private static final Function<Object, Object> DEFAULT_CONTENT_SANITIZER = Function.identity();
35      private static final Function<Throwable, Throwable> DEFAULT_CAUSE_SANITIZER = Function.identity();
36  
37      private LogLevel requestLogLevel = LogLevel.TRACE;
38      private LogLevel successfulResponseLogLevel = LogLevel.TRACE;
39      private LogLevel failedResponseLogLevel = LogLevel.WARN;
40      private Function<? super HttpHeaders, ? extends HttpHeaders> requestHeadersSanitizer =
41              DEFAULT_HEADERS_SANITIZER;
42      private Function<Object, ?> requestContentSanitizer = DEFAULT_CONTENT_SANITIZER;
43      private Function<? super HttpHeaders, ? extends HttpHeaders> responseHeadersSanitizer =
44              DEFAULT_HEADERS_SANITIZER;
45      private Function<Object, ?> responseContentSanitizer = DEFAULT_CONTENT_SANITIZER;
46      private Function<? super Throwable, ? extends Throwable> responseCauseSanitizer = DEFAULT_CAUSE_SANITIZER;
47      private float samplingRate = 1.0f;
48  
49      /**
50       * Sets the {@link LogLevel} to use when logging requests. If unset, will use {@link LogLevel#TRACE}.
51       */
52      public T requestLogLevel(LogLevel requestLogLevel) {
53          this.requestLogLevel = requireNonNull(requestLogLevel, "requestLogLevel");
54          return self();
55      }
56  
57      /**
58       * Returns the {@link LogLevel} to use when logging requests.
59       */
60      protected LogLevel requestLogLevel() {
61          return requestLogLevel;
62      }
63  
64      /**
65       * Sets the {@link LogLevel} to use when logging successful responses (e.g., no unhandled exception).
66       * If unset, will use {@link LogLevel#TRACE}.
67       */
68      public T successfulResponseLogLevel(LogLevel successfulResponseLogLevel) {
69          this.successfulResponseLogLevel =
70                  requireNonNull(successfulResponseLogLevel, "successfulResponseLogLevel");
71          return self();
72      }
73  
74      /**
75       * Returns the {@link LogLevel} to use when logging successful responses (e.g., no unhandled exception).
76       */
77      protected LogLevel successfulResponseLogLevel() {
78          return successfulResponseLogLevel;
79      }
80  
81      /**
82       * Sets the {@link LogLevel} to use when logging failure responses (e.g., failed with an exception).
83       * If unset, will use {@link LogLevel#WARN}. The request will be logged too if it was not otherwise.
84       */
85      public T failureResponseLogLevel(LogLevel failedResponseLogLevel) {
86          this.failedResponseLogLevel = requireNonNull(failedResponseLogLevel, "failedResponseLogLevel");
87          return self();
88      }
89  
90      /**
91       * Returns the {@link LogLevel} to use when logging failure responses (e.g., failed with an exception).
92       */
93      protected LogLevel failedResponseLogLevel() {
94          return failedResponseLogLevel;
95      }
96  
97      /**
98       * Sets the {@link Function} to use to sanitize request headers before logging. It is common to have the
99       * {@link Function} that removes sensitive headers, like {@code Cookie}, before logging. If unset, will use
100      * {@link Function#identity()}.
101      */
102     public T requestHeadersSanitizer(
103             Function<? super HttpHeaders, ? extends HttpHeaders> requestHeadersSanitizer) {
104         this.requestHeadersSanitizer = requireNonNull(requestHeadersSanitizer, "requestHeadersSanitizer");
105         return self();
106     }
107 
108     /**
109      * Returns the {@link Function} to use to sanitize request headers before logging.
110      */
111     protected Function<? super HttpHeaders, ? extends HttpHeaders> requestHeadersSanitizer() {
112         return requestHeadersSanitizer;
113     }
114 
115     /**
116      * Sets the {@link Function} to use to sanitize response headers before logging. It is common to have the
117      * {@link Function} that removes sensitive headers, like {@code Set-Cookie}, before logging. If unset,
118      * will use {@link Function#identity()}.
119      */
120     public T responseHeadersSanitizer(
121             Function<? super HttpHeaders, ? extends HttpHeaders> responseHeadersSanitizer) {
122         this.responseHeadersSanitizer = requireNonNull(responseHeadersSanitizer, "responseHeadersSanitizer");
123         return self();
124     }
125 
126     /**
127      * Returns the {@link Function} to use to sanitize response headers before logging.
128      */
129     protected Function<? super HttpHeaders, ? extends HttpHeaders> responseHeadersSanitizer() {
130         return responseHeadersSanitizer;
131     }
132 
133     /**
134      * Sets the {@link Function} to use to sanitize request and response headers before logging. It is common
135      * to have the {@link Function} that removes sensitive headers, like {@code "Cookie"} and
136      * {@code "Set-Cookie"}, before logging. This method is a shortcut of:
137      * <pre>{@code
138      * builder.requestHeadersSanitizer(headersSanitizer);
139      * builder.responseHeadersSanitizer(headersSanitizer);
140      * }</pre>
141      *
142      * @see #requestHeadersSanitizer(Function)
143      * @see #responseHeadersSanitizer(Function)
144      */
145     public T headersSanitizer(Function<? super HttpHeaders, ? extends HttpHeaders> headersSanitizer) {
146         requireNonNull(headersSanitizer, "headersSanitizer");
147         requestHeadersSanitizer(headersSanitizer);
148         responseHeadersSanitizer(headersSanitizer);
149         return self();
150     }
151 
152     /**
153      * Sets the {@link Function} to use to sanitize request content before logging. It is common to have the
154      * {@link Function} that removes sensitive content, such as an GPS location query, before logging. If unset,
155      * will use {@link Function#identity()}.
156      */
157     public T requestContentSanitizer(Function<Object, ?> requestContentSanitizer) {
158         this.requestContentSanitizer = requireNonNull(requestContentSanitizer, "requestContentSanitizer");
159         return self();
160     }
161 
162     /**
163      * Returns the {@link Function} to use to sanitize request content before logging.
164      */
165     protected Function<Object, ?> requestContentSanitizer() {
166         return requestContentSanitizer;
167     }
168 
169     /**
170      * Sets the {@link Function} to use to sanitize response content before logging. It is common to have the
171      * {@link Function} that removes sensitive content, such as an address, before logging. If unset,
172      * will use {@link Function#identity()}.
173      */
174     public T responseContentSanitizer(Function<Object, ?> responseContentSanitizer) {
175         this.responseContentSanitizer = requireNonNull(responseContentSanitizer, "responseContentSanitizer");
176         return self();
177     }
178 
179     /**
180      * Returns the {@link Function} to use to sanitize response content before logging.
181      */
182     protected Function<Object, ?> responseContentSanitizer() {
183         return responseContentSanitizer;
184     }
185 
186     /**
187      * Sets the {@link Function} to use to sanitize request and response content before logging. It is common
188      * to have the {@link Function} that removes sensitive content, such as an GPS location query and
189      * an address, before logging. If unset, will use {@link Function#identity()}. This method is a shortcut of:
190      * <pre>{@code
191      * builder.requestContentSanitizer(contentSanitizer);
192      * builder.responseContentSanitizer(contentSanitizer);
193      * }</pre>
194      *
195      * @see #requestContentSanitizer(Function)
196      * @see #responseContentSanitizer(Function)
197      */
198     public T contentSanitizer(Function<Object, ?> contentSanitizer) {
199         requireNonNull(contentSanitizer, "contentSanitizer");
200         requestContentSanitizer(contentSanitizer);
201         responseContentSanitizer(contentSanitizer);
202         return self();
203     }
204 
205     /**
206      * Sets the {@link Function} to use to sanitize a response cause before logging. You can
207      * sanitize the stack trace of the exception to remove sensitive information, or prevent from logging
208      * the stack trace completely by returning {@code null} in the {@link Function}. If unset, will use
209      * {@link Function#identity()}.
210      */
211     public T responseCauseSanitizer(Function<? super Throwable, ? extends Throwable> responseCauseSanitizer) {
212         this.responseCauseSanitizer = requireNonNull(responseCauseSanitizer, "responseCauseSanitizer");
213         return self();
214     }
215 
216     /**
217      * Returns the {@link Function} to use to sanitize response cause before logging.
218      */
219     protected Function<? super Throwable, ? extends Throwable> responseCauseSanitizer() {
220         return responseCauseSanitizer;
221     }
222 
223     /**
224      * Sets the rate at which to sample requests to log. Any number between {@code 0.0} and {@code 1.0} will
225      * cause a random sample of the requests to be logged. The random sampling is appropriate for low-traffic
226      * (ex servers that each receive &lt;100K requests). If unset, all requests will be logged.
227      */
228     public T samplingRate(float samplingRate) {
229         checkArgument(0.0 <= samplingRate && samplingRate <= 1.0, "samplingRate must be between 0.0 and 1.0");
230         this.samplingRate = samplingRate;
231         return self();
232     }
233 
234     /**
235      * Returns the rate at which to sample requests to log.
236      */
237     protected float samplingRate() {
238         return samplingRate;
239     }
240 
241     @SuppressWarnings("unchecked")
242     private T self() {
243         return (T) this;
244     }
245 
246     @Override
247     public String toString() {
248         return toString(this, requestLogLevel, successfulResponseLogLevel, failedResponseLogLevel,
249                         requestHeadersSanitizer, requestContentSanitizer, responseHeadersSanitizer,
250                         responseContentSanitizer, samplingRate);
251     }
252 
253     private static <T extends LoggingDecoratorBuilder<T>> String toString(
254             LoggingDecoratorBuilder<T> self,
255             LogLevel requestLogLevel,
256             LogLevel successfulResponseLogLevel,
257             LogLevel failureResponseLogLevel,
258             Function<? super HttpHeaders, ? extends HttpHeaders> requestHeadersSanitizer,
259             Function<?, ?> requestContentSanitizer,
260             Function<? super HttpHeaders, ? extends HttpHeaders> responseHeadersSanitizer,
261             Function<?, ?> responseContentSanitizer,
262             float samplingRate) {
263         final ToStringHelper helper = MoreObjects.toStringHelper(self)
264                                                  .add("requestLogLevel", requestLogLevel)
265                                                  .add("successfulResponseLogLevel", successfulResponseLogLevel)
266                                                  .add("failedResponseLogLevel", failureResponseLogLevel)
267                                                  .add("samplingRate", samplingRate);
268         if (requestHeadersSanitizer != DEFAULT_HEADERS_SANITIZER) {
269             helper.add("requestHeadersSanitizer", requestHeadersSanitizer);
270         }
271         if (requestContentSanitizer != DEFAULT_CONTENT_SANITIZER) {
272             helper.add("requestContentSanitizer", requestContentSanitizer);
273         }
274         if (responseHeadersSanitizer != DEFAULT_HEADERS_SANITIZER) {
275             helper.add("responseHeadersSanitizer", responseHeadersSanitizer);
276         }
277         if (responseContentSanitizer != DEFAULT_CONTENT_SANITIZER) {
278             helper.add("responseContentSanitizer", responseContentSanitizer);
279         }
280         return helper.toString();
281     }
